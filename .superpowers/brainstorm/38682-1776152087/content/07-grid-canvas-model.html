Grid Specification

/* Canvas root */
.workspace-canvas {
display: grid;
grid-template-areas:
"top    top  top"
"left   main right"
"bottom bottom bottom";
grid-template-rows: auto 1fr auto;
grid-template-columns:
var(--left-sheet-width, 0)  /* animated via react-spring */
1fr
var(--right-sheet-width, 0); /* animated via react-spring */
/* Canvas owns the gradient */
background: var(--canvas-gradient);
}

/* Responsive sheet widths */
--left-open:  clamp(260px, 25vw, 360px);
--right-open: clamp(220px, 20vw, 300px);

/* Nav & FAB float freely */
[grid-area="bottom"] { position: relative; }
.navigation-bar { position: absolute; }
.fab { position: absolute; }
